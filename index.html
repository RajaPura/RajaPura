const SHEET_NAME = 'Registos'; function doGet(e) { const action = e.parameter.action; let result; if (action === 'get') { result = getRegistos(); } else { result = { error: 'Ação não reconhecida' }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { try { const data = JSON.parse(e.postData.contents); let result; if (data.action === 'add') { result = addRegisto(data.registo); } else { result = { success: false, error: 'Ação não reconhecida' }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function getRegistos() { try { const sheet = getOrCreateSheet(); const rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return { registos: [] }; const headers = rows[0]; const registos = rows.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => { obj[h] = row[i]; }); try { obj.valores = JSON.parse(obj.valores); } catch(e) { obj.valores = {}; } return obj; }); return { registos }; } catch(err) { return { registos: [], error: err.toString() }; } } function addRegisto(registo) { try { const sheet = getOrCreateSheet(); if (sheet.getLastRow() === 0) { sheet.appendRow(['id', 'nome', 'loja', 'data', 'valores', 'total', 'criado_em']); sheet.getRange(1, 1, 1, 7).setFontWeight('bold').setBackground('#C8913A').setFontColor('#FFFFFF'); sheet.setFrozenRows(1); } sheet.appendRow([ registo.id || Date.now(), registo.nome || '', registo.loja || '', registo.data || '', JSON.stringify(registo.valores || {}), registo.total || 0, new Date().toISOString() ]); return { success: true }; } catch(err) { return { success: false, error: err.toString() }; } } function getOrCreateSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(['id', 'nome', 'loja', 'data', 'valores', 'total', 'criado_em']); sheet.getRange(1, 1, 1, 7).setFontWeight('bold').setBackground('#C8913A').setFontColor('#FFFFFF'); sheet.setFrozenRows(1); } return sheet; }
